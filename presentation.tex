\section{Context}

\begin{frame}{Project Context}
  \begin{block}{Problem}
    Research labs use many kinds of machines, but energy usage is usually invisible.
    Teams can see CPU load, yet they often cannot answer simple questions such as:
    which node wastes power at idle, which application consumes most energy, and how the lab evolves over time.
  \end{block}

  \begin{block}{Our Goal}
    Build one distributed platform that collects energy and health metrics from heterogeneous nodes,
    stores them efficiently, and exposes them through a real-time dashboard.
  \end{block}

  \smallnote{Target environment: workstations, Raspberry Pi devices, and mobile clients.}
\end{frame}

\begin{frame}{What The System Delivers}
  \begin{columns}[T]
    \column{0.5\textwidth}
      \begin{block}{Monitoring Scope}
        \begin{itemize}
          \item 8 system KPIs per node
          \item Per-application energy attribution
          \item Real-time and historical views
          \item Multi-node status tracking
        \end{itemize}
      \end{block}
    \column{0.5\textwidth}
      \begin{block}{Engineering Goals}
        \begin{itemize}
          \item Low-latency reads for dashboards
          \item Reliable ingestion during outages
          \item Database replication for resilience
          \item Clear API-based integration
        \end{itemize}
      \end{block}
  \end{columns}

  \vspace{0.4em}
  \textbf{Core KPIs:} voltage, CPU frequency, CPU utilization, RAM utilization, temperature, power, energy, uptime.
\end{frame}

\section{Used Technologies}

\begin{frame}{Used Technologies}
  \begin{table}
    \small
    \begin{tabular}{>{\raggedright\arraybackslash}p{2.7cm} >{\raggedright\arraybackslash}p{3cm} >{\raggedright\arraybackslash}p{5.3cm}}
      \toprule
      Layer & Technology & Role \\
      \midrule
      Data collection & Python 3.12, psutil, RAPL, INA219 & Collect node KPIs and estimate per-application energy \\
      Backend API & FastAPI, Pydantic, JWT & Validate payloads, expose REST endpoints, manage authentication \\
      Database & ClickHouse, ClickHouse Keeper & Store time-series data, support replication, serve analytical queries \\
      Frontend & React 18, Tailwind CSS, Recharts & Real-time dashboard, historical charts, node and app views \\
      Deployment & Docker Compose, systemd & Run services consistently and deploy daemon on Linux nodes \\
      Benchmarking & Python scripts, synthetic generators & Measure query latency and ingestion throughput \\
      \bottomrule
    \end{tabular}
  \end{table}
\end{frame}

\begin{frame}{Why These Choices}
  \begin{itemize}
    \item \textbf{FastAPI} gives a clean REST layer and automatic API documentation.
    \item \textbf{ClickHouse} fits analytical time-series workloads and supports materialized views.
    \item \textbf{Replication} improves resilience and keeps the storage layer aligned with the project requirements.
    \item \textbf{React + Recharts} is enough for live charts without adding unnecessary complexity.
    \item \textbf{Python daemon} keeps collectors modular and portable across different node types.
  \end{itemize}
\end{frame}

\section{Architecture \& Communication Protocol}

\begin{frame}{Architecture}
  \begin{itemize}
    \item \textbf{Edge layer:} each node runs a daemon that reads local sensors and OS metrics.
    \item \textbf{Service layer:} the backend authenticates requests, validates payloads, updates heartbeat state, and routes reads and writes.
    \item \textbf{Storage layer:} ClickHouse stores raw metrics and materialized views for hourly and daily aggregation.
    \item \textbf{Presentation layer:} the dashboard queries the backend and adapts the aggregation level to the selected time range.
  \end{itemize}

  \vspace{0.4em}
  \begin{block}{Design Logic}
    We separate collection, API logic, storage, and visualization so each layer can evolve independently.
  \end{block}
\end{frame}

\begin{frame}{Communication Protocol}
  \begin{table}
    \small
    \begin{tabular}{>{\raggedright\arraybackslash}p{2.2cm} >{\raggedright\arraybackslash}p{2cm} >{\raggedright\arraybackslash}p{2.3cm} >{\raggedright\arraybackslash}p{4.8cm}}
      \toprule
      Source & Method & Auth & Purpose \\
      \midrule
      Daemon $\rightarrow$ Backend & POST `/api/v1/metrics` & API key & Send timestamped KPI batches and optional app metrics \\
      Dashboard $\rightarrow$ Backend & GET `/api/v1/metrics` & JWT & Read raw or filtered metrics \\
      Dashboard $\rightarrow$ Backend & GET aggregated endpoints & JWT & Read downsampled data for long time ranges \\
      Dashboard $\rightarrow$ Backend & GET `/api/v1/nodes`, `/summary` & JWT & Read node status and system summary \\
      Backend $\rightarrow$ ClickHouse & Native / HTTP client & DB credentials & Insert records and execute analytical queries \\
      \bottomrule
    \end{tabular}
  \end{table}
\end{frame}

\begin{frame}{Data Model and Reliability}
  \begin{columns}[T]
    \column{0.52\textwidth}
      \begin{block}{Data Model}
        Each record is centered on:
        \begin{itemize}
          \item timestamp
          \item node\_id
          \item app\_name
          \item metric
          \item value
        \end{itemize}
        This structure is simple for ingestion and efficient for analytical queries.
      \end{block}
    \column{0.44\textwidth}
      \begin{block}{Reliability Features}
        \begin{itemize}
          \item replicated ClickHouse setup
          \item file-backed retry buffer
          \item adaptive heartbeat tracking
          \item materialized views for faster reads
        \end{itemize}
      \end{block}
  \end{columns}
\end{frame}

\begin{frame}{Example Request Flow}
  \begin{enumerate}
    \item The daemon collects system metrics and app-level energy data.
    \item It sends a JSON payload to the backend with its API key.
    \item The backend validates the payload and stores rows in ClickHouse.
    \item Materialized views prepare hourly and daily aggregates.
    \item The dashboard requests either raw or aggregated data depending on the selected time range.
  \end{enumerate}
\end{frame}
